```typescript
function calculateScores(answers: Answer[]): Scores {
  // Revenue Health (0-100)
  const revenueHealth =
    (answers.revenueGrowth === 'growing' ? 40 :
     answers.revenueGrowth === 'flat' ? 20 : 0) +
    (answers.biggestChallenge !== 'revenue' ? 30 : 0) +
    (answers.incentives ? 30 : 0);

  // Cost Management (0-100)
  const costManagement =
    (answers.costsStatus === 'decreasing' ? 40 :
     answers.costsStatus === 'stable' ? 25 : 0) +
    (answers.biggestChallenge !== 'costs' ? 30 : 0) +
    (answers.tracksMetrics ? 30 : 0);

  // Team Engagement (0-100)
  const teamEngagement =
    (answers.teamEngagement === 'high' ? 40 :
     answers.teamEngagement === 'moderate' ? 20 : 0) +
    (answers.dailyMeetings === 'yes' ? 30 :
     answers.dailyMeetings === 'weekly' ? 15 : 0) +
    (answers.techComfort === 'very' ? 30 :
     answers.techComfort === 'somewhat' ? 15 : 0);

  return { revenueHealth, costManagement, teamEngagement };
}
```
